);

SELECT ID FROM instructor WHERE ID NOT IN (
  SELECT ID FROM teaches;
);

SELECT s.name, c.title, sec.year
FROM student s
JOIN takes t ON s.ID = t.ID
JOIN section sec ON t.course_id = sec.course_id 
     AND t.sec_id = sec.sec_id
     AND t.semester = sec.semester
     AND t.year = sec.year
JOIN course c ON sec.course_id = c.course_id
WHERE sec.room_number = '303';


SELECT s.name, t.course_id

JOIN takes ON s.ID = t.ID
WHERE t.year = 2015;
